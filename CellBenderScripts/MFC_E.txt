#!/bin/bash -l

#SBATCH -N 1 --ntasks-per-node=6 --partition=gpu
#SBATCH --gres=gpu:1
#SBATCH --job-name="PAIR"
#SBATCH --output=/home/dennis00/Yi_lab/MarkerPaper/slurm_archive/MFC_PC_%j.out

echo Hi
module load cuda/11.7

conda activate cellbender

cd $SLURM_SUBMIT_DIR

cd /home/dennis00/Yi_lab/MarkerPaper/CellBenderScripts/MFC/E/

REPS="s01_01 s03_09 s04_09"

for p in $REPS
	do
	cellbender remove-background \
           --input "/home/dennis00/Yi_lab/MarkerPaper/CellBender_RawDataMarkerPaper/MFC/${p}.h5" \
           --output "/home/dennis00/Yi_lab/MarkerPaper/CellBender_ProcessedDataMarkerPaper/MFC/${p}.h5" \
           --cuda \
           --expected-cells 1600 \
           --total-droplets-included 2100 \
           --low-count-threshold 15 \
           --learning-rate 1e-4 \
           --fpr 0.01 \
           --epochs 150
done

echo cellbender_Ma_human
echo "SLURM Job ID        : ${SLURM_JOB_ID}"
